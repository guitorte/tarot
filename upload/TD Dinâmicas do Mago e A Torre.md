1) A engendra B.
2) A em conflito ativo com B.
3) A estagnado diante de B.
4) A reduz-se a B.

def make_prompt(pairs):
    pair_list = "\n".join([f"- {a} (A) e {b} (B)" for a, b in pairs])
    return f"""You are writing a tarot interpretation book in Brazilian Portuguese.

STYLE REFERENCE (follow this exact structure and tone):
{STYLE_EXAMPLE}

For each pair below, write the 5 dynamics in the SAME style as the reference.
Return ONLY a valid JSON array, no markdown, no preamble.
Format: [{{"pair": "CardA x CardB", "text": "...full text..."}}]

PAIRS TO GENERATE:
{pair_list}

DYNAMICS TO COVER FOR EACH PAIR:
{DYNAMICS}
"""

def generate_batch(pairs, retries=3):
    for attempt in range(retries):
        try:
            response = model.generate_content(make_prompt(pairs))
            raw = response.text.strip().lstrip("```json").rstrip("```")
            return json.loads(raw)
        except Exception as e:
            print(f"  Retry {attempt+1}: {e}")
            time.sleep(5)
    return []

# Generate all pairs
all_pairs = list(itertools.product(CARDS, CARDS))  # 78x78
# Or for unique pairs only: itertools.combinations_with_replacement(CARDS, 2)

batches = [all_pairs[i:i+BATCH_SIZE] for i in range(0, len(all_pairs), BATCH_SIZE)]

results = []
for i, batch in enumerate(tqdm(batches)):
    batch_results = generate_batch(batch)
    results.extend(batch_results)
    
    # Save incrementally (don't lose progress)
    with open("tarot_combinations.json", "w", encoding="utf-8") as f:
